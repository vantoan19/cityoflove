```js
let foxFrameTimer = null;
let bunnyFrameTimer = null;

function setPose(character, poseName) {
  const imgEl = document.getElementById(character === 'fox' ? 'foxImg' : 'bunnyImg');
  const base = character === 'fox'
    ? `../chapter1/animations/fox/${poseName}/frame`
    : `../chapter1/animations/bunny/${poseName}/frame`;

  gsap.to(imgEl, { opacity: 0, duration: 0.15, onComplete: () => {
    // Clear old timer
    if (character === 'fox')   { clearInterval(foxFrameTimer);   foxFrameTimer   = null; }
    if (character === 'bunny') { clearInterval(bunnyFrameTimer); bunnyFrameTimer = null; }

    let frame = 1;
    imgEl.src = base + '1.png';
    gsap.to(imgEl, { opacity: 1, duration: 0.15 });

    const timer = setInterval(() => {
      frame = (frame % 5) + 1;
      imgEl.src = base + frame + '.png';
    }, 200);

    if (character === 'fox')   foxFrameTimer   = timer;
    if (character === 'bunny') bunnyFrameTimer = timer;
  }});
}

// Kick off default idle pose
setPose('fox',   'pose1_neutral_idle');
setPose('bunny', 'pose1_neutral_idle');
```

```js
let currentBeat = -1;
let beatTimer = null;
let phase1Done = false;

function advanceBeat() {
  if (phase1Done) return;
  currentBeat++;
  if (currentBeat >= BEATS.length) return;

  const beat = BEATS[currentBeat];

  if (beat.text === null) {
    // Silent beat — fade out then transition
    hideText();
    clearTimeout(beatTimer);
    setTimeout(startPhase2, 1500);
    return;
  }

  showText(beat.text);
  if (beat.fox)   setPose('fox',   beat.fox);
  if (beat.bunny) setPose('bunny', beat.bunny);

  clearTimeout(beatTimer);
  // 0.5s fade-in + 2s hold = 2.5s before auto-advance
  beatTimer = setTimeout(advanceBeat, 2500);
}

// Tap / click
document.addEventListener('click', () => { clearTimeout(beatTimer); advanceBeat(); });

// Scroll
document.addEventListener('wheel', (e) => {
  e.preventDefault();
  clearTimeout(beatTimer);
  advanceBeat();
}, { passive: false });

// Touch swipe up
let touchStartY = 0;
document.addEventListener('touchstart', (e) => { touchStartY = e.touches[0].clientY; }, { passive: true });
document.addEventListener('touchend', (e) => {
  if (touchStartY - e.changedTouches[0].clientY > 30) { clearTimeout(beatTimer); advanceBeat(); }
}, { passive: true });

// Kick off first beat after short delay
setTimeout(advanceBeat, 600);
```

```js
const openedBars = new Set();
let activeMessageEl = null;

function getRenderedRect(img) {
  const iw = img.naturalWidth, ih = img.naturalHeight;
  const cw = img.clientWidth,  ch = img.clientHeight;
  const ir = iw / ih, cr = cw / ch;
  let rw, rh, rx, ry;
  if (ir > cr) { rw = cw; rh = cw / ir; rx = 0; ry = (ch - rh) / 2; }
  else         { rh = ch; rw = ch * ir; ry = 0; rx = (cw - rw) / 2; }
  return { x: rx, y: ry, w: rw, h: rh };
}

function computeBarRect(index, r) {
  const col = index % BAR_GRID.cols;
  const row = Math.floor(index / BAR_GRID.cols);
  const bw  = (BAR_GRID.width  / BAR_GRID.cols) * r.w;
  const bh  = (BAR_GRID.height / BAR_GRID.rows) * r.h;
  return {
    x: r.x + BAR_GRID.left * r.w + col * bw,
    y: r.y + BAR_GRID.top  * r.h + row * bh,
    w: bw, h: bh,
  };
}
```

```js
function initChocGame() {
  const img = document.getElementById('chocBox');
  const container = document.getElementById('barContainer');

  function render() {
    const r = getRenderedRect(img);
    container.innerHTML = '';
    container.style.pointerEvents = 'auto';

    for (let i = 0; i < 8; i++) {
      const br = computeBarRect(i, r);

      const cell = document.createElement('div');
      cell.className = 'bar-cell';
      cell.id = `bar-cell-${i}`;
      cell.style.cssText = `
        position:absolute;
        left:${br.x}px; top:${br.y}px;
        width:${br.w}px; height:${br.h}px;
        overflow:visible;
      `;

      // Unwrap image (hidden until bar is clicked)
      const barImg = document.createElement('img');
      barImg.id = `bar-img-${i}`;
      barImg.style.cssText = 'display:none;position:absolute;inset:0;width:100%;height:100%;object-fit:contain;';
      cell.appendChild(barImg);

      // Shimmer overlay
      const shimmerWrap = document.createElement('div');
      shimmerWrap.style.cssText = 'position:absolute;inset:0;overflow:hidden;pointer-events:none;border-radius:3px;';
      const shimmerBar = document.createElement('div');
      shimmerBar.className = `shimmer-${i}`;
      shimmerBar.style.cssText = `

        background:linear-gradient(90deg,transparent 0%,rgba(255,255,255,0.55) 50%,transparent 100%);
        transform:translateX(-150%) skewX(-20deg);
        pointer-events:none;
      `;
      shimmerWrap.appendChild(shimmerBar);
      cell.appendChild(shimmerWrap);

      // Button
      const btn = document.createElement('button');
      btn.id = `bar-btn-${i}`;
      btn.className = 'bar-btn' + (i === 7 ? ' bar-btn--locked' : '');
      btn.textContent = i === 7 ? '🔒' : 'open';
      btn.dataset.index = i;
      cell.appendChild(btn);

      // Hover
      cell.addEventListener('mouseenter', () => onBarHover(i, true));
      cell.addEventListener('mouseleave', () => onBarHover(i, false));
      btn.addEventListener('click', (e) => { e.stopPropagation(); handleBarClick(i); });

      container.appendChild(cell);
    }
  }

  render();
  window.addEventListener('resize', render);
}
```

```js
const UNWRAP_SUFFIXES = [
  'wrapped',
  'unwrap_2_cornerlift',
  'unwrap_3_topflap45',
  'unwrap_4_topflap90',
  'unwrap_5_fullyopen',
  'unwrap_6_tucked',
];
const UNWRAP_DELAYS_MS = [0, 120, 180, 220, 240, 200];

function runUnwrap(index, onComplete) {
  const cell   = document.getElementById(`bar-cell-${index}`);
  const barImg = document.getElementById(`bar-img-${index}`);
  const btn    = document.getElementById(`bar-btn-${index}`);

  btn.style.display = 'none';
  barImg.style.display = 'block';
  barImg.src = `bars/bar_${index + 1}_wrapped.png`;

  // Lift + wobble
  gsap.timeline()
    .to(cell, { y: -12, rotation: -2, duration: 0.48, ease: 'power2.out' })
    .to(cell, { rotation: 1, duration: 0.24, ease: 'power1.inOut' })
    .to(cell, { y: 0, rotation: 0, duration: 0.24, ease: 'power1.in' });

  // Frame sequence
  let fi = 0;
  function nextFrame() {
    barImg.src = `bars/bar_${index + 1}_${UNWRAP_SUFFIXES[fi]}.png`;
    fi++;
    if (fi < UNWRAP_SUFFIXES.length) {
      setTimeout(nextFrame, UNWRAP_DELAYS_MS[fi]);
    } else {
      setTimeout(onComplete, 250);
    }
  }
  nextFrame();
}
```

```js
  hint.style.opacity = '0';

      btn.classList.remove('bar-btn--locked');

}
```
